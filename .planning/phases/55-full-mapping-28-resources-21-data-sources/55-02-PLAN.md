**SECRETS-03 — TestProviderInfo_AllSensitiveFieldsPromoted:**
Comprehensive map of ALL resource fields with `Sensitive: true` in TF schema. For each, assert `Fields[fieldName].Secret != nil && *Fields[fieldName].Secret == true`. Exception: `flashblade_array_connection_key.id` must be `Secret == false` (acknowledged exposure).

```go
func TestProviderInfo_AllSensitiveFieldsPromoted(t *testing.T) {
    prov := Provider()
    // Every TF field with Sensitive:true must have explicit Secret:tfbridge.True()
    // in resources.go. This test fails loudly if a new sensitive field is added upstream
    // without a corresponding bridge override (SECRETS-03).
    sensitiveFields := map[string][]string{
        "flashblade_array_connection":                    {"connection_key"},
        "flashblade_certificate":                         {"passphrase", "private_key"},
        "flashblade_object_store_access_key":             {"secret_access_key"},
        "flashblade_object_store_remote_credentials":     {"secret_access_key"},
        "flashblade_directory_service_management":        {"bind_password"},
    }
    for resName, fields := range sensitiveFields {
        r, ok := prov.Resources[resName]
        if !ok {
            t.Errorf("resource %q not in Resources", resName)
            continue
        }
        for _, f := range fields {
            info, ok := r.Fields[f]
            if !ok || info == nil || info.Secret == nil || !*info.Secret {
                t.Errorf("resource %q field %q must have Secret=true (SECRETS-03)", resName, f)
            }
        }
    }
}
```
