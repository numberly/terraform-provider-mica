<objective>
Fix R-006: `QosPolicyPost` silently drops `max_total_bytes_per_sec=0` and `max_total_ops_per_sec=0` (which mean "unlimited" on FlashBlade) because of `int64 + omitempty`. Convention §POST struct requires `*int64` when zero has semantic meaning. Also remove the `name` field from JSON body (it goes via `?names=` query param per convention).

Purpose: Preserve "unlimited" semantics (0 value) end-to-end through POST; bring `QosPolicyPost` into convention compliance (matches SubnetPost fix from Phase 51).
Output: Updated model, resource Create builder, schema v0→v1 no-op upgrader, upgrader test.
</objective>

```go
// qosPolicyV0Model mirrors the schema as of Version 0 — identical shape to v1 for this migration.
type qosPolicyV0Model struct {
    ID                  types.String   `tfsdk:"id"`
    Name                types.String   `tfsdk:"name"`
    Enabled             types.Bool     `tfsdk:"enabled"`
    MaxTotalBytesPerSec types.Int64    `tfsdk:"max_total_bytes_per_sec"`
    MaxTotalOpsPerSec   types.Int64    `tfsdk:"max_total_ops_per_sec"`
    IsLocal             types.Bool     `tfsdk:"is_local"`
    PolicyType          types.String   `tfsdk:"policy_type"`
    Timeouts            timeouts.Value `tfsdk:"timeouts"`
}

// UpgradeState returns state upgraders for schema migrations.
func (r *qosPolicyResource) UpgradeState(ctx context.Context) map[int64]resource.StateUpgrader {
    return map[int64]resource.StateUpgrader{
        0: {
            PriorSchema: &schema.Schema{
                Version: 0,
                Attributes: map[string]schema.Attribute{
                    // EXACT copy of v0 schema — same attributes as v1 (POST-only migration).
                    "id":                      schema.StringAttribute{Computed: true},
                    "name":                    schema.StringAttribute{Required: true},
                    "enabled":                 schema.BoolAttribute{Optional: true, Computed: true},
                    "max_total_bytes_per_sec": schema.Int64Attribute{Optional: true},
                    "max_total_ops_per_sec":   schema.Int64Attribute{Optional: true},
                    "is_local":                schema.BoolAttribute{Computed: true},
                    "policy_type":             schema.StringAttribute{Computed: true},
                    "timeouts": timeouts.Attributes(ctx, timeouts.Opts{
                        Create: true, Read: true, Update: true, Delete: true,
                    }),
                },
            },
            StateUpgrader: func(ctx context.Context, req resource.UpgradeStateRequest, resp *resource.UpgradeStateResponse) {
                var old qosPolicyV0Model
                resp.Diagnostics.Append(req.State.Get(ctx, &old)...)
                if resp.Diagnostics.HasError() {
                    return
                }
                newState := qosPolicyModel{
                    ID:                  old.ID,
                    Name:                old.Name,
                    Enabled:             old.Enabled,
                    MaxTotalBytesPerSec: old.MaxTotalBytesPerSec,
                    MaxTotalOpsPerSec:   old.MaxTotalOpsPerSec,
                    IsLocal:             old.IsLocal,
                    PolicyType:          old.PolicyType,
                    Timeouts:            old.Timeouts,
                }
                resp.Diagnostics.Append(resp.State.Set(ctx, newState)...)
            },
        },
    }
}
```
