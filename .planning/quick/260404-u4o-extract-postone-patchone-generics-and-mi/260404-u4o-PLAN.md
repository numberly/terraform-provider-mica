func getOneByName[T any](c *FlashBladeClient, ctx context.Context, path, label, name string) (*T, error) {
    var resp ListResponse[T]
    if err := c.get(ctx, path, &resp); err != nil {
        return nil, err
    }
    if len(resp.Items) == 0 {
        return nil, &APIError{StatusCode: 404, Message: fmt.Sprintf("%s %q not found", label, name)}

}

func postOne[TBody any, TResp any](c *FlashBladeClient, ctx context.Context, path string, body TBody, label string) (*TResp, error) {

        return nil, fmt.Errorf("%s: empty response from server", label)
